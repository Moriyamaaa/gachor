Reset Password via Email (EmailJS)





// Ganti dengan PUBLIC KEY dari EmailJS let generatedOTP = ""; function sendEmail() { const uname = document.getElementById("username").value; const email = document.getElementById("email").value; const users = JSON.parse(localStorage.getItem("users") || "[]"); const user = users.find(u => u.username === uname); if (!user) { alert("User tidak ditemukan!"); return; } generatedOTP = Math.floor(100000 + Math.random() * 900000).toString(); const params = { to_email: email, to_name: uname, otp_code: generatedOTP }; emailjs.send(SERVICE_ID, TEMPLATE_ID, params) .then(() => { alert("OTP dikirim ke email!"); document.getElementById("otpSection").style.display = "block"; }, err => { alert("Gagal mengirim email: " + JSON.stringify(err)); }); } function verifyOTP() { const inputOTP = document.getElementById("otpInput").value; const newPass = document.getElementById("newPassword").value; const uname = document.getElementById("username").value; if (inputOTP !== generatedOTP) { alert("OTP salah!"); return; } let users = JSON.parse(localStorage.getItem("users") || "[]"); const idx = users.findIndex(u => u.username === uname); users[idx].password = newPass; localStorage.setItem("users", JSON.stringify(users)); alert("Password berhasil direset."); window.location.href = "login.html"; }
